\frametitle{Rappel du sujet}
\begin{enumerate}
    \item Rappeler le fonctionnement des marchés de taux.
    \item Construire des courbes déterministes sur la base des taux échangés par maturités liquides.
    \item Prolonger les courbes de taux sur les maturités manquantes ou peu liquides.
    \item Proposer une approche stochastique pour projeter les courbes de taux en probabilité historique.
\end{enumerate}

    \frametitle{un autre modèle classique : Nelson-Siegel-Svensson}
    \begin{block}{Modèle de Nielson-Siegel 1987} 
        \[
        f(t)
        = \beta_0
        + \beta_1 e^{-t/\tau_1}
        + \beta_2 \frac{t}{\tau_1} e^{-t/\tau_1}
        \]
    \end{block}

    \begin{block}{Modèle de Nielson-Siegel-Svensson 1994} 
        \[
        f(t)
        = \beta_0
        + \beta_1 e^{-t/\tau_1}
        + \beta_2 \frac{t}{\tau_1} e^{-t/\tau_1}
        + \beta_3 \frac{t}{\tau_2} e^{-t/\tau_2}
        \]
    \end{block}

\end{frame}
